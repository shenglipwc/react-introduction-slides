We can also pass the callback through props

class App extends React.Component {
	state = { loginUser: null }
	handleLogin = (username) => {
		this.setState({ loginUser: username })
	}
	render() {
		return <Navigation onLogin={this.handleLogin} />
	}
}
class Navigation extends React.Component {
	onClick = ()=> this.props.handleLogin("default")
	render() {
		return (<button onClick={this.onClick}>login</button>)
	}
}
